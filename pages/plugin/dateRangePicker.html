Date Range Picker

참고 사이트 링크
자바스크립트 라이브러리 2가지와 (daterangepicker.js, moment.js)
CSS 라이브러리 1가지를 추가 후 진행 (daterangepicker.css)

필요한 파일 
// CSS
<link rel="stylesheet" href="https://static.goodchoice.kr/static/ui/css/library/daterange/daterangepicker.css">

// JS 순서 유의
<script src="https://static.goodchoice.kr/static/ui/js/library/daterange/moment.min.js"></script>
<script src="https://static.goodchoice.kr/static/ui/js/library/daterange/daterangepicker.js"></script>
싱글형 
<div class="input-group mb-2" style="width: 200px">
    <div class="input-group-prepend">
        <i class="icon-baseline-calendar_today-24px"></i>
    </div>
    <input type="text" class="form-control datePickerTest01">
</div>
$('.datePickerTest01').daterangepicker({
    singleDatePicker: true, 
});
기본 멀티형 
<div class="input-group mb-2" style="width: 300px">
    <div class="input-group-prepend">
        <i class="icon-baseline-calendar_today-24px"></i>
    </div>
    <input type="text" class="form-control datePickerTest02">
</div>
$('.datePickerTest02').daterangepicker({
    autoApply: true,
});
