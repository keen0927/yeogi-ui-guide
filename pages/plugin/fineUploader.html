3. 실행 스크립트를 추가 하세요.

var manualUploader = new qq.FineUploader({
    element: document.getElementById('fine-uploader-manual-trigger'), // HTML ELEMENT
    template: 'qq-template-manual-trigger',
    request: {
        endpoint: '/server/uploads' // 엔드 포인트
    },
    thumbnails: {
        placeholders: {
            waitingPath: 'https://static.goodchoice.kr/static/ui/js/library/fine-uploader/images/waiting-generic.png',
            notAvailablePath: 'https://static.goodchoice.kr/static/ui/js/library/fine-uploader/images/not_available-generic.png'
        }
    },
    autoUpload: false,
    debug: true,
    validation: {
        allowedExtensions: ['jpeg', 'jpg', 'txt', 'png'],
        itemLimit: 3,
        sizeLimit: 5120 // 5000 kB = 50 * 1024 bytes
    }
});

qq(document.getElementById("trigger-upload")).attach("click", function() {
    manualUploader.uploadStoredFiles();
});
