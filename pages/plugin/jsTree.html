JsTree

플러그인 사이트 : https://www.jstree.com/

jsTree는 jQuery기반으로 HTML, JSON 데이터를 Tree 컨트롤로 웹에 출력해 주는 자바스크립트 라이브러리 입니다.

플러그인 추가
기본 CSS와 JS를 추가 해 주세요. (CSS는 head안에 JS는 jQuery하단에 위치하도록 정의) 
// CSS
https://static.goodchoice.kr/static/ui/css/library/jsTree/style.min.css

// JS
https://static.goodchoice.kr/static/ui/js/library/jsTree/jstree.min.js
HTML 데모
  • 루트 노드
    • 자식 노드 1
    • 자식 노드 2
<div id="html" class="demo">
    <ul>
        <li data-jstree='{ "opened" : true }'>루트 노드
            <ul>
                <li data-jstree='{ "selected" : true }'>자식 노드 1</li>
                <li>자식 노드 2</li>
            </ul>
        </li>
    </ul>
</div>

$('#html').jstree();
인라인 데이타 데모 
$('#data').jstree({
    'core' : {
        'data' : [
            { "text" : "Root node", "children" : [
                    { "text" : "Child node 1" },
                    { "text" : "Child node 2" }
            ]}
        ]
    }
});
데이타 포맷 데모 
$('#frmt').jstree({
    'core' : {
        'data' : [
            {
                "text" : "Root node",
                "state" : { "opened" : true },
                "children" : [
                    {
                        "text" : "Child node 1",
                        "state" : { "selected" : true },
                        "icon" : "jstree-file"
                    },
                    { "text" : "Child node 2", "state" : { "disabled" : true } }
                ]
            }
        ]
    }
});

Lazy Load 데모 
$('#lazy').jstree({
    'core' : {
        'data' : {
            "url" : "//www.jstree.com/fiddle/?lazy",
            "data" : function (node) {
                return { "id" : node.id };
            }
        }
    }
});
콜백 데모 
$('#clbk').jstree({
    'core' : {
        'data' : function (node, cb) {
            if(node.id === "#") {
                cb([{"text" : "Root", "id" : "1", "children" : true}]);
            }
            else {
                cb(["Child"]);
            }
        }
    }
});
콜백 데모 
$('#evts_button').on("click", function () {
    var instance = $('#evts').jstree(true);
    instance.deselect_all();
    instance.select_node('1');
});
$('#evts')
    .on("changed.jstree", function (e, data) {
        if(data.selected.length) {
            alert('선택된 노드는 : ' + data.instance.get_node(data.selected[0]).text);
        }
    })
    .jstree({
        'core' : {
            'multiple' : false,
            'data' : [
                { "text" : "Root node", "children" : [
                        { "text" : "Child node 1", "id" : 1 },
                        { "text" : "Child node 2" }
                ]}
            ]
        }
    });
