샘플용 Validate
아이디는 필수, 비밀번호는 필수, 특정메소드 추가, 최소길이8, 최대길이20을 적용한 샘플.

required값을 true로 설정하면 필수 값으로 인식
minlength와 maxlength의 값도 적용하여 지정해준 값이 아닌경우 통과할 수 없다.

커스텀 validate
샘플코드에는 정규식을 활용해 영문, 숫자, 특수문자를 포함하도록 하는 정규식을 사용 하였다.
사용방법
첫번째 매개변수에 사용할 커스텀 네이밍을 지정해주고 뒤의 콜백 함수에 필요한 정규식을 추가하면 된다.

// 커스텀 메소드
$.validator.addMethod("passwordCheck", function (value, element) {
    return this.optional(element) || /^(?=.*[a-zA-Z])(?=.*[#\&\\+\-%@=\/\\\:;,\.\'\"\^`~\_|\!\/\?\*$#<>()\[\]\{\}])(?=.*[0-9])/.test(value);
});

// 기본형
$("#loginForm").validate({
    rules: { // 룰스에 필요한 validate을 지정
        labelId: { // 선택자(labelId)
            required: true,
        },
        labelPassword: { // 선택자(labelPassword)
            required: true,
            passwordCheck: true, // 커스텀메소드가 트루일때 통과
            minlength: 8,
            maxlength: 20,
        }
    },
    messages: { // validate 오류시 노출해줄 메세지를 설정
        labelId: "아이디를 입력해 주세요.",
        labelPassword: {
            required: "비밀번호를 입력해 주세요.",
            passwordCheck: "영문,숫자,특수문자를 포함한 8~20자로 입력해주세요.",
            minlength: '영문,숫자,특수문자를 포함한 8~20자로 입력해주세요.',
            maxlength: '영문,숫자,특수문자를 포함한 8~20자로 입력해주세요.',
        }
    }
});
비동기통신 Validate
서버와의 통신이 필요한경우 다음과 같이 remote 메소드를 사용할 수 있다. [공식문서] 
// 이메일 정규식 체크
$.validator.addMethod("emailCheck", function( value, element ) {
    return this.optional( element ) || /[0-9a-zA-Z][_0-9a-zA-Z-]*@[_0-9a-zA-Z-]+(\.[_0-9a-zA-Z-]+){1,2}$/.test( value );
});

$('셀렉터').validate({
    rules: {
        email: {
            required: true,
            email: true,
            remote: "/email/check" // 확인할 경로
        }
    },
    messages: {
        email: {
            required: "이메일을 입력해주세요.",
            emailCheck: "올바른 이메일주소를 입력해주세요."
            remote: "이미 등록된 이메일 주소입니다."
        }
    }
});
정규식 샘플

휴대폰번호 체크 정규식

// 휴대폰
$.validator.addMethod("phonenumber", function( value, element ) {
    return this.optional( element ) || /^(01[016789]{1}|02|0[3-9]{1}[0-9]{1})-?([0-9]{3,4})-?([0-9]{4})$/.test( value );
});

한글,영문,숫자 체크

// 한글,영문,숫자 체크
$.validator.addMethod("nameId", function( value, element ) {
    return this.optional( element ) || /^[0-9|가-힣|a-z|A-Z\*]+$/.test( value );
});
