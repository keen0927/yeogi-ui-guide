체크박스

체크박스 샘플

1. var checkbox = new CheckBox(targetElement, options) 로 셀렉트박스를 생성한다.

2. checkbox.create()하여 DOM에 그린다.



// 테스트 코드
var checkboxTarget1 = $('#checkbox1');
var examValues = [
	{ value: 'bnn', text: 'banana', checked: false, disable: false },
	{ value: 'app', text: 'apple', checked: true, disable: false },
	{ value: 'grp', text: 'grape', checked: false, disable: true },
	{ value: 'tmt', text: 'tomato', checked: false, disable: false },
];

// 체크박스 객체 생성 (필수)
var checkbox1 = new CheckBox(checkboxTarget1, {
	attributes: { name: 'name', class: 'mr-2' },
	values: examValues,
});

// DOM에 그리는 시점 (필수)
checkbox1.create();

// 사용자가 셀렉트한 값 가져오기
$('#getSelectedValue').on('click', function() {
	alert(checkbox1.getCheckedValue());
});

// 모든 체크박스 체크하기
$('#checkAll').on('click', function() {
	checkbox1.checkAll();
});

// 모든 체크박스 체크 해제하기
$('#uncheckAll').on('click', function() {
	checkbox1.unCheckAll();
});
API 설명
checkbox.getCheckedValue() 사용자가 선택한 value를 가져온다.
checkbox.checkAll() 모든 체크박스를 체크한다.
checkbox.unCheckAll(); 모든 체크박스를 체크 해제한다.
